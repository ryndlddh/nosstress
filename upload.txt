<?php
// Proses jika form telah di-submit
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    // Ambil nilai input dari form
    $photo_id = $_POST['photo_id'];
    $user_id = $_POST['user_id'];
    $album_id = $_POST['album_id'];
    $title = $_POST['title'];
    $description = $_POST['description'];
    $image_path = $_POST['image_path'];
    $create_at = $_POST['create_at'];

    // Koneksi ke database
    $conn = mysqli_connect("localhost", "root", "", "album");

    // Cek koneksi
    if (!$conn) {
        die("Koneksi gagal: " . mysqli_connect_error());
    }

    // Query untuk insert data ke dalam tabel photos
    $sql = "INSERT INTO photos (photo_id, user_id, album_id, title, description, image_path, create_at)
            VALUES ('$photo_id', '$user_id', '$album_id', '$title', '$description', '$image_path', '$create_at')";

    if (mysqli_query($conn, $sql)) {
        echo "Data berhasil ditambahkan.";
    } else {
        echo "Error: " . $sql . "<br>" . mysqli_error($conn);
    }

    // Tutup koneksi ke database
    mysqli_close($conn);
}
?>

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Create Data</title>
</head>
<body>
    <h2>Form Create Data</h2>
    <form method="post" action="<?php echo htmlspecialchars($_SERVER["PHP_SELF"]); ?>">
        <label for="photo_id">Photo ID:</label><br>
        <input type="text" id="photo_id" name="photo_id"><br>
        <label for="user_id">User ID:</label><br>
        <input type="text" id="user_id" name="user_id"><br>
        <label for="album_id">Album ID:</label><br>
        <input type="text" id="album_id" name="album_id"><br>
        <label for="title">Title:</label><br>
        <input type="text" id="title" name="title"><br>
        <label for="description">Description:</label><br>
        <textarea id="description" name="description"></textarea><br>
        <label for="image_path">Image Path:</label><br>
        <input type="text" id="image_path" name="image_path"><br>
        <label for="create_at">Create At:</label><br>
        <input type="text" id="create_at" name="create_at"><br><br>
        <input type="submit" value="Submit">
    </form>
</body>
</html>
